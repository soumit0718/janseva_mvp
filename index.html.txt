// ── BUILD ISSUE CARD HTML ─────────────────────────────────────
function buildIssueCard(iss){
  var isAdmin = currentUser && currentUser.role !== 'citizen';
  var bodyHtml='';
  if(iss.expanded){
    var comments='';
    if(iss.comments.length>0){
      var chtml='';
      iss.comments.forEach(function(c){
        chtml+='<div style="margin-bottom:6px"><div class="update-author">'+c.author+' <span class="update-date">• '+c.date+'</span></div><div class="update-text">'+c.text+'</div></div>';
      });
      comments='<div class="updates-box"><div class="updates-title">UPDATES</div>'+chtml+'</div>';
    }
    var adminBtns='';
    if(isAdmin){
      var btns='';
      ['Open','In Progress','Under Review','Resolved'].forEach(function(s){
        var active=iss.status===s;
        btns+='<button class="status-btn" style="background:'+(active?'#1B2A4A':'#F1F5F9')+';color:'+(active?'#fff':'#334155')+'" onclick="updateStatus(\''+iss.id+'\',\''+s+'\')">'+s+'</button>';
      });
      adminBtns='<div class="status-btns">'+btns+'</div>';
    }
    bodyHtml='<div class="issue-body"><div class="issue-desc">'+iss.description+'</div>'+(iss.assignedTo?'<div class="issue-assigned">👷 Assigned to: '+iss.assignedTo+'</div>':'')+comments+adminBtns+'</div>';
  }
  return '<div class="issue-card" id="card-'+iss.id+'">'+
    '<div class="issue-header" onclick="toggleIssue(\''+iss.id+'\')">'+
      '<div style="display:flex;justify-content:space-between;align-items:flex-start;gap:10px">'+
        '<div style="flex:1">'+
          '<div class="issue-badges">'+badge(iss.status)+badge(iss.priority)+badge(iss.level)+'</div>'+
          '<div class="issue-title">'+iss.title+'</div>'+
          '<div class="issue-meta">📍 '+iss.location+' • 📅 '+iss.date+' • '+iss.id+'</div>'+
        '</div>'+
        '<div class="issue-votes"><div class="vnum">'+iss.upvotes+'</div><div class="vlabel">VOTES</div></div>'+
      '</div>'+
    '</div>'+bodyHtml+
  '</div>';
}

function submitIssue(){
  var title=document.getElementById('f-title').value.trim();
  var location=document.getElementById('f-location').value.trim();
  var desc=document.getElementById('f-desc').value.trim();
  if(!title||!location||!desc){ showToast('Please fill all required fields','#E53E3E'); return; }
  var newId='ISS'+String(issues.length+1).padStart(3,'0');
  issues.unshift({
    id:newId, title:title, location:location, description:desc,
    category:document.getElementById('f-category').value,
    priority:document.getElementById('f-priority').value,
    level:document.getElementById('f-level').value,
    status:'Open', submittedBy:currentUser.name,
    date:new Date().toISOString().slice(0,10),
    upvotes:1, comments:[], assignedTo:null, expanded:false
  });
  document.getElementById('f-title').value='';
  document.getElementById('f-location').value='';
  document.getElementById('f-desc').value='';
  closeModal();
  renderDashboard(); renderIssues();
  showToast('✅ Issue reported! ID: '+newId);
}

function getFallback(msg){
  var l=msg.toLowerCase();
  if(l.includes('complaint')||l.includes('issue')||l.includes('problem'))
    return 'To file a complaint:\n1. Click "Report Issue" button\n2. Select category (Water, Road, Electricity etc.)\n3. Describe the problem clearly\n4. Submit — you\'ll get a tracking ID\n\nYour complaint goes directly to the concerned officer!';
  if(l.includes('scheme')||l.includes('yojana')||l.includes('awas'))
    return 'For PM Awas Yojana:\nDocuments needed:\n• Aadhaar card\n• Income certificate (below ₹1.8 lakh)\n• Land ownership proof\n• Bank account details\n• Ration card\n\nApply at your nearest GP or Municipal office. Application is FREE!';
  if(l.includes('certificate')||l.includes('birth')||l.includes('document'))
    return 'For Birth Certificate:\nRequired documents:\n• Hospital discharge slip\n• Parent\'s Aadhaar cards\n• Marriage certificate of parents\n\nTimeline: 7-15 working days\nFee: ₹10 only\n\nApply at eGramSwaraj portal or your ward office.';
  if(l.includes('water')||l.includes('pipeline')||l.includes('tank'))
    return 'For water supply complaints:\n1. File complaint with your address\n2. Mention duration of disruption\n\nEmergency: Call 1916 (Municipal helpline)\n\nIf unresolved in 48 hours, escalate to Municipal Commissioner.';
  return 'I can help with complaints, government schemes, certificates, and local governance queries. Could you be more specific about what you need help with?';
}

async function sendChat(){
  var input=document.getElementById('chat-input');
  var msg=input.value.trim();
  if(!msg)return;
  input.value='';
  chatMessages.push({role:'user',text:msg});
  chatMessages.push({role:'ai',text:'Thinking…'});
  renderChatMessages();
  var reply=getFallback(msg);
  try{
    var res=await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-20250514',max_tokens:400,
        system:'You are Jan Sahayak, an AI assistant for Indian local governance (Gram Panchayat and Municipality). Help citizens in simple, friendly language. Topics: complaints, government schemes, certificates, water, roads, meetings. Keep responses under 120 words. Use bullet points. Be empathetic. Mix Hindi words naturally like Namaste, Ji.',
        messages:[{role:'user',content:msg}]
      })
    });
    if(res.ok){var d=await res.json();reply=d.content&&d.content[0]?d.content[0].text:reply;}
  }catch(e){}
  chatMessages[chatMessages.length-1]={role:'ai',text:reply};
  renderChatMessages();
}
